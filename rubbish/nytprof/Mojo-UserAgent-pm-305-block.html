464 $req->url($url->base($self->app_url)->to_abs) unless $url->host;
465 }
466
467 # Proxy
468 $self->detect_proxy if $ENV{MOJO_PROXY};
469 my $req = $tx->req;
470 my $url = $req->url;
471 my $scheme = $url->scheme || '';
472 if ($self->need_proxy($url->host)) {
473
474 # HTTP proxy
475 if (my $proxy = $self->http_proxy) {
476 $req->proxy($proxy) if !$req->proxy && $scheme eq 'http';
477 }
478
479 # HTTPS proxy
480 if (my $proxy = $self->https_proxy) {
481 $req->proxy($proxy) if !$req->proxy && $scheme eq 'https';
482 }
483 }
484
485 # We identify ourselves
486 my $headers = $req->headers;
487 $headers->user_agent($self->name) unless $headers->user_agent;
488

490 if (my $jar = $self->cookie_jar) { $jar->inject($tx) }
491
492 # Connect
493 $self->emit(start => $tx);
494 return unless my $id = $self->_connect($tx, $cb);
495 $self->{processing} += 1;
496
497 # Request timeout
498 if (my $t = $self->request_timeout) {
499 weaken $self;
500 my $loop = $self->_loop;
501 $self->{connections}->{$id}->{timeout} =
502 $loop->timer($t => sub { $self->_error($id, 'Request timeout.') });

508sub _upgrade {
509 my ($self, $id) = @_;
510
511 # No upgrade request
512 my $c = $self->{connections}->{$id};
513 my $old = $c->{tx};
514 return unless $old->req->headers->upgrade;
515
516 # Handshake failed
517 my $res = $old->res;
518 return unless ($res->code || '') eq '101';
519
520 # Upgrade to WebSocket transaction
521 my $new = Mojo::Transaction::WebSocket->new(handshake => $old, masked => 1);
522 $new->kept_alive($old->kept_alive);
523 $res->error('WebSocket challenge failed.') and return
524 unless $new->client_challenge;
525 $c->{tx} = $new;
526 $self->_loop->stream($id)->timeout($self->websocket_timeout);
527 weaken $self;
528 $new->on(resume => sub { $self->_write($id) });
529
530 return $new;
531}

533sub _write {
534 my ($self, $id) = @_;
535
536 # Prepare outgoing data
537 return unless my $c = $self->{connections}->{$id};
538 return unless my $tx = $c->{tx};
539 return unless $tx->is_writing;
540 return if $self->{writing}++;
541 my $chunk = $tx->client_write;
542 delete $self->{writing};
543 warn "> $chunk\n" if DEBUG;
544
545 # More data to follow
546 my $cb;
547 if ($tx->is_writing) {
548 weaken $self;
549 $cb = sub { $self->_write($id) };
550 }
551
552 # Write data
553 $self->_loop->stream($id)->write($chunk, $cb);
554 $self->_handle($id) if $tx->is_finished;
555}
