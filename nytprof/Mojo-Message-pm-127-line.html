443sub _parse_formdata {
444 my $self = shift;
445
446 # Check content
447 my @formdata;
448 my $content = $self->content;
449 return \@formdata unless $content->is_multipart;

451
452 # Walk the tree
453 my @parts;
454 push @parts, $content;
455 while (my $part = shift @parts) {
456
457 # Multipart
458 if ($part->is_multipart) {
459 unshift @parts, @{$part->parts};
460 next;
461 }
462

464 my $charset = $part->charset || $default;
465
466 # Content-Disposition header
467 my $disposition = $part->headers->content_disposition;
468 next unless $disposition;

470 my ($filename) = $disposition =~ /\ filename="?([^"]*)"?/;
471 my $value = $part;
472
473 # Unescape
474 $name = url_unescape $name if $name;
475 $filename = url_unescape $filename if $filename;
476 if ($charset) {

478 $filename = decode($charset, $filename) // $filename if $filename;
479 }

486 if $charset && !$part->headers->content_transfer_encoding;
